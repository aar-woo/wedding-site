      import { createHmac, timingSafeEqual } from 'node:crypto';

      export function encodePayload(payload) {
        // payload is { id, name, iat }
        return Buffer.from(JSON.stringify(payload), 'utf8').toString('base64url');
      }

      function hmac(base64urlPayload, secret) {
        return createHmac('sha256', secret).update(base64urlPayload).digest('base64url');
      }

      export function sign(payload, secret) {
        // payload MUST be { id, name, iat }. Returns the `t=` value: `${b64Payload}.${b64Hmac}`.
        const b64 = encodePayload(payload);
        return `${b64}.${hmac(b64, secret)}`;
      }

      export function verify(token, secret) {
        // returns true ONLY if token is well-formed AND the hmac matches. Never throws.
        if (typeof token !== 'string') return false;
        const parts = token.split('.');
        if (parts.length !== 2) return false;
        const [b64, sig] = parts;
        if (!b64 || !sig) return false;
        const expected = hmac(b64, secret);
        const sigBuf = Buffer.from(sig, 'base64url');
        const expBuf = Buffer.from(expected, 'base64url');
        if (sigBuf.length !== expBuf.length) return false;   // length check before timingSafeEqual
        return timingSafeEqual(sigBuf, expBuf);              // timing-safe comparison (NOT ===)
      }

    Notes:

    - This file is Node-only and secret-bearing; the secret is passed in as an argument (callers read

      export function decodeGuestToken(token) {
        // token is the `t=` query value: `<base64url-payload>.<base64url-hmac>`
        // Returns { id, name } on success, or null on missing/malformed/undecodable input. Never throws.
        try {
          if (typeof token !== 'string' || token.length === 0) return null;
          const b64 = token.split('.')[0];
          if (!b64) return null;
          // browser-safe base64url -> JSON. Convert base64url to base64, then atob, then UTF-8 decode.
          const base64 = b64.replace(/-/g, '+').replace(/_/g, '/');
          const json = decodeURIComponent(escape(atob(base64)));  // UTF-8-safe atob
          const payload = JSON.parse(json);
          if (!payload || typeof payload.id !== 'string' || typeof payload.name !== 'string') return null;
          return { id: payload.id, name: payload.name };
        } catch {
          return null;
        }
      }
